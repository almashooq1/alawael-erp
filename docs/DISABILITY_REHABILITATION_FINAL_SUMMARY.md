## 🚀 كيفية البدء | Getting Started

```javascript
// استيراد الخدمات
const { AdvancedPhysicalTherapyService } = require('./advanced-physical-therapy-service');
const { UnifiedAssessmentService } = require('./unified-assessment-service');
const { CaseManagementService } = require('./case-management-service');

// إنشاء مثيلات
const ptService = new AdvancedPhysicalTherapyService();
const assessmentService = new UnifiedAssessmentService();
const caseService = new CaseManagementService();

// إنشاء حالة جديدة
const case_ = await caseService.createCase({
  beneficiaryId: 'BEN-001',
  type: 'physical_rehabilitation',
  priority: 'high'
});

// إجراء تقييم
const assessment = await assessmentService.createAssessment('BEN-001', {
  type: 'comprehensive',
  domains: ['physical', 'cognitive', 'social', 'vocational']
});
```
